34       private void sendRequestWithHttp(){ 
35           new Thread(new Runnable() { 
36               @Override 
37               public void run() { 
38                   try{ 
39                       OkHttpClient client=new OkHttpClient();//步骤1，实例化客户端client 
40                       Request request=new Request.Builder()//步骤2，实例化请求Request 
41                               //指定服务器是本机 
42                               .url("Http://192.168.1.102/httpgetxmldata.xml") 
43                               .build(); 
44                       Response response=client.newCall(request).execute();//步骤3，用response接收返回 
45                       String responseData=response.body().string();//步骤4，将返回的response里面的Body转化为String，以便接下来的提取 
46                       parseXmlUsePull(responseData); 
47                   }catch (Exception e){ 
48                       e.printStackTrace(); 
49                   } 
50               } 
51           }).start();

54       private void parseXmlUsePull(String xmlData){ 
55           try{ 
56               XmlPullParserFactory xmlPullParserFactory=XmlPullParserFactory.newInstance(); 
57               XmlPullParser xmlPullParser=xmlPullParserFactory.newPullParser(); 
58               xmlPullParser.setInput(new StringReader(xmlData)); 
59               int eventType=xmlPullParser.getEventType(); 
60               String id=""; 
61               String name=""; 
62               String version=""; 
63               int count=0; 
64               while(eventType!=xmlPullParser.END_DOCUMENT){ 
65                   String nodeName=xmlPullParser.getName(); 
66                   switch(eventType){ 
67                       case XmlPullParser.START_TAG:{ 
68                           if("id".equals(nodeName)){ 
69                               id=xmlPullParser.nextText(); 
70                           }else if("name".equals(nodeName)){ 
71                               name= xmlPullParser.nextText(); 
72                           }else if ("version".equals(nodeName)){ 
73                               version=xmlPullParser.nextText(); 
74                           } 
75                           break; 
76                       } 
77                       case XmlPullParser.END_TAG:{ 
78                           if ("app".equals(nodeName)){ 
79                               Log.d("huan","id is:"+id); 
80                               Log.d("huan","name is:"+name); 
81                               Log.d("huan","version is:"+version); 
82                           } 
83                           break; 
84                       } 
85                       default: 
86                           break; 
87                   } 
88                   eventType=xmlPullParser.next(); 
89                   count++; 
90                   Log.d("count",String.valueOf(count)); 
91               } 
92           }catch (Exception e){ 
93               e.printStackTrace(); 
94           } 
95       }
